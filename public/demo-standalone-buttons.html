üìö Usage Guide

1. Include the CSS

<link rel="stylesheet" href="./css/bootstrap.css" />
<link rel="stylesheet" href="./css/buttons.css" />

2. Use the Classes

<!-- Solid buttons -->
<button class="btn btn-blue">Blue Button</button>
<button class="btn btn-green">Green Button</button>

<!-- Outline buttons -->
<button class="btn btn-outline-purple">Purple Outline</button>

<!-- Soft buttons -->
<button class="btn btn-soft-pink">Pink Soft</button>

<!-- Combine with Bootstrap sizing -->
<button class="btn btn-lg btn-indigo">Large Indigo</button>
<button class="btn btn-sm btn-red">Small Red</button>
